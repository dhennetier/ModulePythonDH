Programme deadline.py 
  Saisir un objectif et une date limite pour cet objectif
  En sortie : 
      informer du nombre de jour qu il reste pour atteindre cet objectif

import datetime
# pour voir la documentation sur le module 
# https://docs.python.org/3/library/datetime.html 

def interligne():
    print("-"*60 + "\n")
    
interligne()    
var_entree=input("Entrer un objectif et une date separe par :\n") 
# exemple Apprendre Python:31.12.2021
list_var_entree=var_entree.split(":") 
objectif=list_var_entree[0] 
dateobj=list_var_entree[1]

# Convertion de dateobj de string DD.MM.YY en date
# on aura le format 2021-12-12 00:00:00 et le type  
fmtdate_obj=datetime.datetime.strptime(dateobj,"%d.%m.%Y")

"""
On va maintenant calculer  combien de jours il nous reste jusqu' a la deadline 

datetime.datetime.today()  : donne la date du jour 
cela donne 2021-12-14 12:32:41.839814  pour le 14 décembre 2021 

"""
fmtdate_jour=datetime.datetime.today()
fmtjours= fmtdate_obj - fmtdate_jour 
# exemple fmtjours  : 16 days, 11:12:57.106096  
# pas propre : on va extraire le nombre de jours avec la builtin fonction .days
jours=fmtjours.days
heures=int(fmtjours.total_seconds()/3600) 

interligne()
print(f"Pour atteindre votre objectif : {objectif} \n")
print(f"\tIl vous reste : {jours} jours \n ") 
print(f"\tIl vous reste : {heures} heures \n ") 
interligne()
